 223 " Compile function
 224 noremap j :call CompileRunGcc()<CR>
 225 func! CompileRunGcc()
 226   exec "w"
 227   if &filetype == 'c'
 228     set splitbelow
 229     :sp
 230     :res -5
 231     term gcc % -o %< && time ./%<
 232   elseif &filetype == 'cpp'
 233     set splitbelow
 234     exec "!g++ -std=c++11 % -Wall -o %<"
 235     :sp
 236     :res -15
 237     :term ./%<
 238   elseif &filetype == 'cs'
 239     set splitbelow
 240     silent! exec "!mcs %"
 241     :sp
 242     :res -5
 243     :term mono %<.exe
 244   elseif &filetype == 'java'
 245     set splitbelow
 246     :sp
 247     :res -5
 248     term javac % && time java %<
 249   elseif &filetype == 'sh'
 250     :!time bash %
 251   elseif &filetype == 'python'
 252     set splitbelow
 253     :sp
 254     :term python3 %
 255   elseif &filetype == 'markdown'
 256     exec "InstantMarkdownPreview"
 257   elseif &filetype == 'tex'
 258     silent! exec "VimtexStop"
 259     silent! exec "VimtexCompile"
 260   elseif &filetype == 'javascript'
 261     set splitbelow
 262     :sp
 263     :term export DEBUG="INFO,ERROR,WARNING"; node --trace-warnings .
 264   elseif &filetype == 'racket'
 265     set splitbelow
 266     :sp
 267     :res -5
 268     term racket %
 269   elseif &filetype == 'go'
 270     set splitbelow
 271     :sp
 272     :term go run .
 273   endif
 274 endfunc
